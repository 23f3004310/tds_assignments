# Generates department distribution, counts R&D frequency, and saves a histogram.

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

def generate_synthetic_employee_data(n=100, seed=42):
    rng = np.random.default_rng(seed)
    departments = ["R&D", "IT", "Finance", "Operations", "Sales", "HR", "Marketing"]
    regions = ["Africa", "Europe", "Middle East", "North America", "South America", "Asia", "Oceania"]
    df = pd.DataFrame({
        "employee_id": [f"EMP{str(i+1).zfill(3)}" for i in range(n)],
        "department": rng.choice(departments, size=n, p=[0.18,0.18,0.14,0.16,0.14,0.10,0.10]),
        "region": rng.choice(regions, size=n),
        "performance_score": np.round(rng.normal(75, 10, size=n).clip(40, 100), 2),
        "years_experience": rng.integers(0, 21, size=n),
        "satisfaction_rating": np.round(rng.normal(4.0, 0.7, size=n).clip(1.0, 5.0), 2),
    })
    return df

# Try to load employees.csv if present
import os
if os.path.exists("employees.csv"):
    df = pd.read_csv("employees.csv")
else:
    df = generate_synthetic_employee_data(100)

rd_count = int((df["department"] == "R&D").sum())
print(f'Frequency count for "R&D" department: {rd_count}')

sns.set_style("whitegrid")
sns.set_context("talk")

fig, ax = plt.subplots(figsize=(8,6), dpi=100)
order = df["department"].value_counts().index.tolist()
sns.countplot(data=df, x="department", order=order, ax=ax, palette="deep")
ax.set_title("Employee Count by Department")
ax.set_xlabel("Department")
ax.set_ylabel("Count")
for label in ax.get_xticklabels():
    label.set_rotation(30)
    label.set_ha("right")

plt.tight_layout()
fig.savefig("department_histogram.png")
